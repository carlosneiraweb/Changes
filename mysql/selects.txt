select u.nick as nick, prov.nombre as provincia, p.fechaPost as fecha, p.titulo as titulo, img.ruta as ruta, p.titulo as titulo, p.comentario as comentario, tc.tiempo as tiempoCambio
from usuario AS u, post AS p, imagenes AS img, provincias AS prov, direccion as dir, tiempo_cambio as tc
where p.idUsuario = u.idUsuario and p.idPost = 77 and img.post_idPost = 77
and dir.provincias_idprovincias = prov.idprovincias 
and tc.idTiempoCambio = p.tiempo_cambio_idTiempoCambio  limit 1
